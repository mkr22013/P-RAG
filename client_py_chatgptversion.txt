def resolve_insurance_topic(query_words, full_query_text):
    """SINGLE SOURCE OF TRUTH: Maps keywords to our specific index topics."""
    topics = []
    query_lower = full_query_text.lower()

    # 1. DEDUCTIBLES/oop (Keyword Match)
    if any(
        w in query_words
        for w in ["deductible", "deductibles", "limit", "oop", "coinsurance"]
    ):
        if "pocket" in query_lower or "oop" in query_lower:
            topics.append("out_of_pocket")
        topics.append("deductible")

    # 2. EMERGENCY & AMBULANCE (Surgical Match)
    # \ber\b ensures 'ER' is a standalone word. We also check for 'room'.
    if (
        re.search(r"\ber\b", query_lower)
        or "emergency" in query_lower
        or "ambulance" in query_words
        or "room" in query_words
    ):
        # If 'urgent' is also present, we pivot to Urgent Care topic
        if "urgent" in query_words:
            topics.append("urgent")
        topics.append("emergency")

    # 3. URGENT CARE (Dedicated Topic)
    if any(w in query_words for w in ["urgent", "clinic", "after-hours"]):
        topics.append("urgent")

    # 4. IMAGING / DIAGNOSTIC (Splitting the logic)
    if any(w in query_words for w in ["xray", "x-ray", "blood", "diagnostic"]):
        topics.append("diagnostic")

    if any(w in query_words for w in ["mri", "pet", "scan", "imaging", "ct"]):
        topics.append("imaging")

    # 5. NETWORK & PROVIDER (Network is a common topic that doesn't fit neatly into the others)
    if any(
        w in query_words for w in ["network", "provider", "pay less", "balance-billing"]
    ):
        topics.append("network")

    # 6. DENTAL & VISION
    if any(w in query_words for w in ["dental", "ortho", "braces"]):
        topics.append("orthodontia")
    if any(w in query_words for w in ["vision", "eye", "glasses"]):
        topics.append("vision")

    # 7. PRIMARY & SPECIALIST
    if (
        any(
            w in query_words
            for w in ["pcp", "primary", "doctor", "physician", "copay", "specialist"]
        )
        or "primary care" in query_lower
    ):
        if "specialist" in query_words:
            topics.append("specialist")
        topics.append("primary")

    # 8. MENTAL HEALTH & SUBSTANCE ABUSE
    if any(
        w in query_words
        for w in ["mental", "behavioral", "substance", "abuse", "psychiatrist"]
    ):
        topics.append("mental-health")

    # 9. PREGNANCY & CHILDBIRTH
    if any(
        w in query_words
        for w in ["pregnant", "maternity", "childbirth", "delivery", "prenatal"]
    ):
        topics.append("maternity")

    # 10. HOSPITAL & SKILLED NURSING
    if any(w in query_words for w in ["hospital", "inpatient", "nursing", "facility"]):
        topics.append("hospital")

    # 11. CHILDREN'S VISION & DENTAL
    if "child" in query_words or "children" in query_words:
        if any(w in query_words for w in ["vision", "eye", "glasses"]):
            topics.append("child-vision")
        if any(w in query_words for w in ["dental", "teeth", "check-up"]):
            topics.append("child-dental")

    # 12. PHARMACY & DRUGS
    if any(
        w in query_words
        for w in ["drug", "prescription", "pharmacy", "generic", "brand", "specialty"]
    ):
        topics.append("pharmacy")

    # 13. REHAB & HABILITATIVE (Common in Premera SBCs)
    if any(
        w in query_words
        for w in [
            "rehab",
            "rehabilitation",
            "habilitative",
            "therapy",
            "physical",
            "speech",
            "occupational",
        ]
    ):
        topics.append("rehabilitation")

    print(f"[*] Resolved Topics from query : {query_lower}  : {topics}")
    return topics if topics else ["benefit"]


def squeeze_insurance_data(raw_text):
    if not raw_text:
        return ""

    # 1. REMOVE URLs + junk
    text = re.sub(r"https?://\S+|For more information.*|&amp;", " ", raw_text)

    # 2. Normalize whitespace
    text = text.replace("\t", " ")
    text = re.sub(r"\s{2,}", " ", text)

    # 🚨 3. STOP if already processed (PREVENT RECURSION)
    if "[||]" in text:
        already_structured = True
    else:
        already_structured = False

    # 4. SBC flip fix
    text = re.sub(r"(\$[\d,]+[^\?]+)\s+([^?]+\?)", r"\2 [||] \1", text)

    # 5. Add structure ONLY if raw
    if not already_structured:
        text = re.sub(
            r"([A-Za-z][A-Za-z\s]{2,}?)\s+(\$[\d,]+|[\d]+%|Not covered|No charge)",
            r"\1 [||] \2",
            text,
            flags=re.IGNORECASE,
        )

    # 6. Normalize pipes ONCE
    text = re.sub(r"\|+", "[||]", text)

    # 7. HARD CLEAN recursive brackets (STRONG FIX)
    for _ in range(3):
        text = re.sub(r"\[\s*\[\s*", "[", text)
        text = re.sub(r"\]\s*\]\s*", "]", text)

    # 8. Remove duplicates
    text = re.sub(r"(\[\|\|\]\s*){2,}", "[||] ", text)

    # 9. Clean lines
    lines = []
    for line in text.split("\n"):
        line = line.strip()
        if not line:
            continue
        lines.append(re.sub(r"\s{2,}", " ", line))

    return "\n".join(lines).strip()


def lock_plan_metadata(
    found_years,
    detected_type,
    detected_tier,
    detected_product=None,
    detected_variant=None,
    detected_network=None,
):
    """
    SOT LOCK: Surgical identity matching against the 6-way Lock.
    Ensures Sherwood HMO (2026) is prioritized over Standard PPO (2024/25).
    """
    from server import get_available_plans
    import ast

    raw_schema = get_available_plans()
    if "DATABASE INFO" in raw_schema:
        return (
            found_years,
            detected_type,
            detected_tier,
            "Standard",
            "Standard",
            "Standard Network",
        )

    try:
        schema_data = ast.literal_eval(
            raw_schema.replace(
                "DATA SOURCE SCHEMA (Year, Type, Tier, Product, Variant, Network): ", ""
            )
        )
    except Exception as e:
        print(f"[*] SCHEMA PARSE ERROR: {e}")
        return (
            found_years,
            detected_type,
            detected_tier,
            "Standard",
            "Standard",
            "Standard Network",
        )

    # 1. NORMALIZE TARGETS
    # found_years might be ['2026'] or '2026', we need a single string for comparison
    search_year = (
        str(found_years[0]) if isinstance(found_years, list) else str(found_years)
    )
    d_type = str(detected_type or "Medical").lower()
    d_tier = str(detected_tier or "Gold").lower()
    d_net = str(detected_network or "").lower()
    d_prod = str(detected_product or "").lower()

    best_match = None

    # 2. THE SURGICAL SEARCH
    # We loop through all available plans to find the closest semantic match for this YEAR
    for row in schema_data:
        y, ty, tr, p, v, n = row

        # Must match Year, Type (Medical/Dental), and Tier (Gold/Silver)
        if str(y) == search_year and ty.lower() == d_type and tr.lower() == d_tier:

            # IDENTITY HANDSHAKE: Match "HMO" to "Sherwood HMO" or vice-versa
            # This prevents 2026 HMO data from using 2024 PPO defaults
            net_match = (d_net in n.lower() or n.lower() in d_net) if d_net else True
            prod_match = (
                (d_prod in p.lower() or p.lower() in d_prod) if d_prod else True
            )

            if net_match and prod_match:
                best_match = row
                break

    # 3. FALLBACK (If specific variant match fails, stay in the correct YEAR silo)
    if not best_match:
        print(
            f"[*] Variant mismatch. Defaulting to first available {search_year} plan."
        )
        best_match = next(
            (
                r
                for r in schema_data
                if str(r[0]) == search_year
                and r[1].lower() == d_type
                and r[2].lower() == d_tier
            ),
            (search_year, d_type, d_tier, "Standard", "Standard", "Standard Network"),
        )

    _, ty, tr, p, v, n = best_match
    print(f"[*] IDENTITY LOCKED: {search_year} | {ty} {tr} | {p} | {n}")

    return (found_years, ty, tr, p, v, n)

def flatten_message_content(content):
    """
    NUCLEAR NORMALIZER: Forces any Ollama response (List, Dict, or None)
    into a plain string to prevent Gradio/Streamlit/Pydantic validation errors.
    """
    if not content:
        return ""
    if isinstance(content, str):
        return content
    if isinstance(content, list):
        parts = []
        for item in content:
            if isinstance(item, dict):
                parts.append(item.get("text", str(item)))
            else:
                parts.append(str(item))
        return " ".join(parts).strip()
    return str(content).strip()


def extract_topic_blocks(chunk, topic_keywords):
    lines = chunk.split("[||]")

    blocks = []
    current_block = []

    for line in lines:
        l = line.strip()

        # 🚀 start new block when topic keyword appears
        if any(k in l.lower() for k in topic_keywords):
            if current_block:
                blocks.append(" [||] ".join(current_block))
                current_block = []

            current_block.append(l)
            continue

        # continue existing block
        if current_block:
            current_block.append(l)

    if current_block:
        blocks.append(" [||] ".join(current_block))

    return blocks

def get_topic_keywords(topic):
    # ✅ Normalize topic safely
    if isinstance(topic, list):
        topic = topic[0]

    topic = str(topic).lower().strip()

    raw_keywords = TOPIC_SYNONYMS.get(topic, [topic])

    # ✅ FLATTEN + CLEAN
    flat_keywords = []

    for k in raw_keywords:
        if isinstance(k, list):
            flat_keywords.extend([str(x) for x in k])
        else:
            flat_keywords.append(str(k))

    print(
        "✅ FINAL KEYWORDS:",
        flat_keywords,
        "| TYPES:",
        [type(x) for x in flat_keywords],
    )

    return flat_keywords


def matches_topic(first_col, topic_keywords):
    try:
        # ✅ FORCE STRING
        first_col = str(first_col)

        words = set(re.findall(r"\w+", first_col.lower()))

        keyword_words = set()

        for k in topic_keywords:
            try:
                if isinstance(k, list):
                    for sub in k:
                        keyword_words.update(str(sub).lower().split())
                else:
                    keyword_words.update(str(k).lower().split())
            except Exception as e:
                print("❌ KEYWORD ERROR:", k, "| TYPE:", type(k), "| ERROR:", e)

        return bool(words & keyword_words)

    except Exception as e:
        print("❌ matches_topic FAILED:", e)
        return False


def parse_structured_rows(merged_topic_data, topic_keywords):
    """
    Robust state-machine parser for [||]-delimited insurance data
    """

    # 🔥 STEP 1: Normalize (kill all formatting noise)
    stream = str(merged_topic_data).replace("\n", " ").replace("  ", " ").strip()

    tokens = stream.split("[||]")

    rows = []
    buffer = []

    def is_valid_cell(cell):
        if not cell:
            return False
        cell = cell.strip().lower()

        # 🚫 junk filters
        if len(cell) < 2:
            return False
        if cell in {"none", "n/a"}:
            return False

        return True

    def looks_like_cost(cell):
        cell = cell.lower()
        return (
            "%" in cell
            or "$" in cell
            or "covered" in cell
            or "not covered" in cell
            or "no charge" in cell
            or "coinsurance" in cell
            or "copay" in cell
        )

    i = 0
    while i < len(tokens):
        cell = str(tokens[i]).strip()

        if not is_valid_cell(cell):
            i += 1
            continue

        buffer.append(cell)

        # 🔥 STATE: Try forming a row when we have 3 cells
        if len(buffer) >= 3:
            benefit = buffer[0]
            inn = buffer[1]
            oon = buffer[2]

            # ✅ HARD VALIDATION (this is the magic)
            if (
                matches_topic(benefit.lower(), topic_keywords)
                and looks_like_cost(inn)
                and looks_like_cost(oon)
            ):
                rows.append(f"{benefit} | {inn} | {oon}")
                buffer = []
            else:
                # 🔥 SHIFT WINDOW (state machine behavior)
                buffer.pop(0)

        i += 1

    return rows
